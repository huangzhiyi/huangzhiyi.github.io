【实验目的】

  • 掌握数据库、SpringMVC、ORM框架、Thymeleaf 综合开发。

【实验环境】

  • Windows 7 以上64位操作系统

【实验资源】

  • JDK 8 - Java Development Kit是 Oracle 公司针对Java开发人员发布的免费软件开发工具包，是 Java 开发必备的开发工具。
  • Intellij IDEA - 业界简称IDEA，是 jetbrains 公司推出的和 Eclipse 齐名的 Java 集成开发环境（IDE）。
  • Apache Maven - Apache Maven，是一个软件项目管理及自动构建工具，由Apache软件基金会所提供。是 Java 构建打包最广泛使用的工具。
  • MySQL 世界上最受欢迎的开源关系数据库之一。
  • HeidiSQL，开源数据库客户端。
  • Cmder 是一个美化的 Windows 命令行终端，支持 ctrl-V 粘贴。
  • Notepad++ 是一个开源文本编辑器，支持代码着色等多种功能，比 Windows 原生文本编辑器功能要丰富很多。
  • Flyway 是一个管理数据库脚本迁移的工具。
  • Eclipse Mosquitto 是一个开源的 MQTT 服务器。支持5.0、3.1.1和3.1的MQTT协议。
  • MQTT.fx 是一个基于 JavaFX 的 MQTT 客户端.

@Controller
public class TempConfigController {
    @Autowired
    TempConfigService tempConfigService;

    @Autowired
    EmojiService emojiService;

    //省略之前代码...

    /**
     * 编辑单个 TempConfig
     * @param model
     * @return
     */
    @GetMapping("/tempconfig/{id}")
    public String edit( @PathVariable("id") String id, Model model){
        TempConfig tempconfig=tempConfigService.getById(Long.valueOf(id));
        model.addAttribute("tempconfig",tempconfig);
        model.addAttribute("emojis",emojiService.getAll());
        return "tempconfig-edit";
    }

    /**
     * 查看 TempConfig 列表
     * @param model
     * @return
     */
    @GetMapping("/tempconfiglist")
    public String list(Model model){
        List<TempConfig> tempconfigs=tempConfigService.getAll();
        model.addAttribute("tempconfiglist"  ,tempconfigs);
        return "tempconfiglist";
    }

    /**
     * 更新 TempConfig
     * @param tempConfig
     * @param model
     * @return
     */
    @PostMapping("/tempconfig-update")
    public String update(TempConfig tempConfig,Model model){
        boolean succ=tempConfigService.update(tempConfig);
        if(succ){
            model.addAttribute("msg","更新成功");
        }else{
            model.addAttribute("msg","更新失败");
        }
        model.addAttribute("tempconfig",tempConfig);
        model.addAttribute("emojis",emojiService.getAll());
        return "tempconfig-edit";
    }

    /**
     * 创建 TempConfig
     * @param tempConfig
     * @param model
     * @return
     */
    @PostMapping("/tempconfig-create")
    public String create(TempConfig tempConfig,Model model){
        boolean succ=tempConfigService.create(tempConfig);
        if(succ){
            model.addAttribute("msg","创建成功");
        }else{
            model.addAttribute("msg","创建失败");
        }
        model.addAttribute("tempconfig",tempConfig);
        model.addAttribute("emojis",emojiService.getAll());
        return "tempconfig-create";
    }

    /**
     * 创建 TempConfig 的表单
     * @param model
     * @return
     */
    @GetMapping("/tempconfig-form")
    public String form(Model model){
        model.addAttribute("emojis",emojiService.getAll());
        return "tempconfig-create";
    }

    /**
     * 删除 TempConfig
     * @param id
     * @param model
     * @return
     */
    @GetMapping("/tempconfig-delete")
    public String delete(@RequestParam(name = "id") Long id, Model model){
        boolean succ=tempConfigService.deleteById(id);
        if(succ){
            model.addAttribute("msg","删除成功");
        }else{
            model.addAttribute("msg","删除失败");
        }
        List<TempConfig> tempconfigs=tempConfigService.getAll();
        model.addAttribute("tempconfiglist"  ,tempconfigs);
        return "tempconfiglist";
    }
}

  • TempConfigServiceImpl新增方法定义
@Service
public class TempConfigServiceImpl implements TempConfigService{

    @Autowired
    TempConfigDao tempConfigDao;

    @Autowired
    EmojiTempMapper emojiTempMapper;

	//省略之前定义的方法...
	@Override
    public TempConfig getById(Long id) {
        return tempConfigDao.findById(id).get();
    }

    @Override
    public List<TempConfig> getAll() {
        List<TempConfig> result=new LinkedList<TempConfig>();
        Iterable<TempConfig> tcs=tempConfigDao.findAll();
        tcs.forEach(e -> result.add(e));
        return result;
    }

    @Override
    public boolean update(TempConfig tc) {
        return emojiTempMapper.update(tc)>0;
    }

    @Override
    public boolean create(TempConfig tc) {
        return emojiTempMapper.create(tc)>0;
    }

    @Override
    public boolean deleteById(Long id) {
        return emojiTempMapper.delete(id)>=0;
    }
}
